
# Commands
~~~bash
# start server
./bin/monit start

# stop server
./bin/monit stop

# restart server
./bin/monit restart

# print helps
./bin/monit -h

# print version
./bin/monit -v

# reload all proxy servers config
./bin/monit reload

# reset service state
./bin/monit reset

# print server status
./bin/monit status
~~~
